             String fieldName,
             Type fieldType,
             Class fieldClass,

             Class itemClass,
             int ordinal,
             long features,
             String format,
             Locale locale,

                 fieldClass,
                 itemType,
                 itemClass,
                 ordinal,
                 features,
                 format,
                 locale,
                 defaultValue,
                 schema,
                 null,
                 field,
                 null
         );
         fieldOffset = UnsafeUtils.objectFieldOffset(field);
     }
 
     @Override
     public void accept(Object object, Object value) {
         if (schema != null) {
             schema.assertValidate(value);
         }
 
         UNSAFE.putObject(object, fieldOffset, value);
     }
