             int year,
             int month,
             int dom,
             int hour,
             int minute,
             int second
     ) {
         final int DAYS_PER_CYCLE = 146097;
         final long DAYS_0000_TO_1970 = (DAYS_PER_CYCLE * 5L) - (30L * 365L + 7L);
 
         long total = (365 * year)
                 + ((year + 3) / 4 - (year + 99) / 100 + (year + 399) / 400)
                 + ((367 * month - 362) / 12)
                 + (dom - 1);
 
         if (month > 2) {
             total--;

             if (leapYear == false) {
                 total--;
             }
         }
 
         long epochDay = total - DAYS_0000_TO_1970;
         return epochDay * 86400
                 + hour * 3600
                 + minute * 60
                 + second;
