     private final boolean defaultState;
     private final int mask;
 
     SerializationFeature(boolean defaultState) {
         this.defaultState = defaultState;
         mask = (1 << ordinal());
     }
 
     @Override
     public boolean enabledByDefault() {
         return defaultState;
     }
 
     @Override
     public int getMask() {
         return mask;
     }
 
     @Override
     public boolean enabledIn(int flags) {
         return (flags & mask) != 0;
     }
