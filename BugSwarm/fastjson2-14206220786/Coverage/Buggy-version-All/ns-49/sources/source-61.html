             if (jsonWriter.isEnabled(NullAsDefaultValue.mask | WriteNullListAsEmpty.mask)) {
                 jsonWriter.startArray();

                 jsonWriter.writeNull();

             }

         }
         jsonWriter.endArray();
     }

     }
