             return false;
         }

             long msb2 = UUIDUtils.parse4Nibbles(str, 4);

             long lsb3 = UUIDUtils.parse4Nibbles(str, 24);
             long lsb4 = UUIDUtils.parse4Nibbles(str, 28);
 
             return (msb1 | msb2 | msb3 | msb4 | lsb1 | lsb2 | lsb3 | lsb4) >= 0;
         }

     }
